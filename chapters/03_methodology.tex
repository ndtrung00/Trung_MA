\chapter{Methodology}\label{chapter:methodology}

\section{Research Design}\label{sec:meth:design}

Five configurations are evaluated against 11 frontier LLMs on 81 long CUAD contracts
($>$100{,}000 chars), yielding $5 \times 11 = 55$ official runs and 163{,}644
analysed predictions. Differences between configurations are minimised to a single
architectural axis per comparison (Table~\ref{tab:configs}), following the controlled
comparative protocol of \cite{dror2018hitchhikers,card2020little}. Every statistical
test is paired on the intersection of sample IDs so performance differences are
attributable to architecture or model, not dataset drift.

\section{Dataset and Sampling}\label{sec:meth:data}

CUAD~\cite{hendrycks2021cuad} is the evaluation corpus (see Table~\ref{tab:cuad-stats}
for statistics). We restrict to 81 contracts $>$100{,}000 characters because
long-context performance is the regime where decomposition is hypothesised to add
value, and long contracts approximate the operational distribution of CLM platforms.
The 81-contract subset yields $81 \times 41 = 3{,}321$ raw (contract, category) pairs
per run.

\paragraph{Sampling strategy.}\label{sec:meth:sampling}
A \texttt{neg\_ratio=0.7} sampler produces a natural 30\%/70\% positive/negative
split per run, mirroring CUAD's actual class prior. Prior work with inflated
positive ratios ($\approx\!67\%$) masks over-extraction because few absent instances
are evaluated; the natural prior exposes it directly. Sampling is seeded
(\texttt{seed=42}) so the same (contract, category) pairs are evaluated across all
configurations, making every statistical test paired at the instance level.
Each run yields $\approx\!2{,}999$ predictions; 1{,}297 of 164{,}941 total
($0.79\%$) were excluded for malformed classification tokens.

\paragraph{Prevalence tiers.}
The 41 CUAD categories are partitioned into three tiers --- \emph{common},
\emph{moderate}, \emph{rare} --- by natural positive rate in the 81-contract subset.
Tier cutpoints are fixed before any model is run.

\section{Metrics}\label{sec:meth:metrics}

All metrics are defined in Section~\ref{sec:lit:eval} (Equations~\ref{eq:fbeta}--\ref{eq:overextract}).
$F_2$ is the primary hypothesis-testing metric ($\beta{=}2$, recall weighted $4\times$
over precision). $F_1$ is reported as a secondary comparison-friendly metric.
Jaccard similarity (Eq.~\ref{eq:jaccard}) measures span quality conditional on
extraction. Laziness ($\lambda$, Eq.~\ref{eq:laziness}) and over-extraction
($\omega$, Eq.~\ref{eq:overextract}) are diagnostic rates. All metrics are
reported pooled and per prevalence tier.

\section{Hypotheses}\label{sec:meth:hypotheses}

\begin{description}[leftmargin=2.5em, labelwidth=2em, itemsep=3pt]
  \item[\textbf{H1.}] $\exists\, C \in \{\text{M0, M1, M6}\}$ such that $C$
    outperforms $B^\star \in \{\text{B1, B4}\}$ on $F_2$. Null: no such $C$ exists.
  \item[\textbf{H2.}] $\Delta\mathrm{Gap}(C_{\text{multi}}, B^\star) < 0$ where
    $\mathrm{Gap}(C) = F_2^{\text{common}}(C) - F_2^{\text{rare}}(C)$.
    Null: $\Delta\mathrm{Gap} \geq 0$.
  \item[\textbf{H3.}] M1 and M6 are statistically distinguishable on $F_2$.
    Null: $\Delta F_2(\text{M1},\text{M6}) = 0$.
\end{description}

\section{Statistical Methodology}\label{sec:meth:stats}

\paragraph{McNemar's exact test.}
For configurations $A$ and $B$ evaluated on the same instances, let
$b = |\{i : A_i \text{ correct},\, B_i \text{ wrong}\}|$ and
$c = |\{i : A_i \text{ wrong},\, B_i \text{ correct}\}|$.
The two-sided exact $p$-value is:
%
\begin{equation}\label{eq:mcnemar}
  p = 2\,\mathbb{P}\!\left[\mathrm{Bin}\!\left(b{+}c,\,\tfrac{1}{2}\right)
      \leq \min(b,c)\right]
\end{equation}
%
implemented via \texttt{scipy.stats.binomtest}~\cite{virtanen2020scipy}. The exact
form is preferred over the $\chi^2$ approximation because discordant counts are small
on rare-tier subsets~\cite{mcnemar1947note}.

\paragraph{Paired bootstrap.}
For continuous metrics ($F_2$, $F_1$, Jaccard, gap differences), Algorithm~\ref{alg:bootstrap}
gives the paired bootstrap CI~\cite{efron1979bootstrap,riezler2005pitfalls}.

\begin{algorithm}[h]
  \caption{Paired bootstrap confidence interval for $\Delta F_2$}
  \label{alg:bootstrap}
  \begin{algorithmic}[1]
    \Require Predictions $\hat{y}^A$, $\hat{y}^B$, labels $y$, $B{=}300$, $\alpha$
    \For{$b = 1$ \textbf{to} $B$}
      \State $\mathcal{I} \gets$ sample $n$ indices with replacement from $\{1,\dots,n\}$
      \State $\delta_b \gets F_2(\hat{y}^A_{\mathcal{I}}, y_{\mathcal{I}}) -
                           F_2(\hat{y}^B_{\mathcal{I}}, y_{\mathcal{I}})$
    \EndFor
    \State \Return $\left[\mathrm{percentile}(\delta, \alpha/2),\;
                          \mathrm{percentile}(\delta, 1{-}\alpha/2)\right]$
  \end{algorithmic}
\end{algorithm}

\paragraph{Multiple-comparison correction.}
The H1 family contains $\binom{5}{2} = 10$ pairwise comparisons per model.
Benjamini--Hochberg~\cite{benjamini1995controlling} controls FDR at $\alpha{=}0.05$:
reject $H_0^{(i)}$ if
%
\begin{equation}\label{eq:bh}
  p_{(i)} \leq \frac{i}{m}\,\alpha,
  \quad p_{(1)} \leq \cdots \leq p_{(m)},\; m = 10.
\end{equation}
%
Raw and BH-corrected $p$-values are both reported. Intersection of sample IDs is taken
symmetrically when partial runs differ in coverage; intersection size is reported with
each test.

\section{Threats to Validity}\label{sec:meth:threats}

\paragraph{Internal.} Configuration parity is maintained by holding the CUAD loader,
sampling seed, prompt registry, and trace schema constant across configurations. M1
and M6 share the same textual specialist instructions (split vs.\ concatenated), so
M1 vs.\ M6 isolates architecture, not prompt content.

\paragraph{External.} Single language (English), single jurisdiction (U.S.\ commercial
contracts), single corpus (CUAD). Model snapshots are pinned; provider updates can
shift behaviour.

\paragraph{Construct.} $F_2$ is a proxy for legal review utility, not a direct
measurement. Reviewer cost of inspecting a spurious span is workflow-dependent; this
is mitigated by reporting precision, recall, $\lambda$, and $\omega$ separately.

\paragraph{Statistical.} Bootstrap percentile CIs under-cover for small samples and
skewed statistics~\cite{efron1979bootstrap}; $p$-values are treated as decisive only
when corroborated by the McNemar exact test. Model identity is a fixed factor;
inference does not generalise beyond the 11 models studied.
