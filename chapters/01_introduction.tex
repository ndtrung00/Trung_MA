\chapter{Introduction}\label{chapter:introduction}

Organisations forfeit $\approx\!9\%$ of annual revenue through poor contracting
practices~\cite{worldcc2025benchmark}; 83\% of executives report contracts too rigid
to accommodate change, with data dispersed across 24 systems on
average~\cite{worldcc2025benchmark}. Goldman Sachs estimate 44\% of U.S.\ legal-task
time is technically automatable by generative AI --- second-highest exposure across all
professions~\cite{goldmansachs2023ai}. Frontier LLMs match junior lawyers on
issue-identification F-scores ($0.87$ vs.\ $0.88$) while reducing review time by
$\approx\!100\times$~\cite{martin2024bettercallgpt}, and independent benchmarks
confirm competitiveness on standardised clause-extraction
tasks~\cite{hendrycks2021cuad,liu2025contracteval}. Single-prompt deployments
nonetheless dominate commercial CLM, and production failures --- hallucination, prompt
sensitivity~\cite{chatterjee2024posix}, and
laziness~\cite{liu2025contracteval} --- remain difficult to localise and remediate.

Multi-agent architectures are the proposed remedy: decomposing clause analysis into
specialised cooperating agents to improve recall, explainability, and regulatory
compliance under the EU~AI~Act~\cite{euaiact2024} and
GDPR~\cite{gdpr2016}. Frameworks including AutoGen~\cite{wu2023autogen},
MetaGPT~\cite{hong2024metagpt}, and
Mixture-of-Agents~\cite{wang2024moa} make such architectures cheap to assemble.
Yet existing legal multi-agent systems --- ChatLaw~\cite{cui2024chatlaw},
PAKTON~\cite{raptopoulos2025pakton}, L-MARS~\cite{wang2025lmars} --- each
evaluate a single architecture on a single benchmark without ablating the architectural
choice itself. Cross-domain evidence questions whether multi-agent designs help at all
once base-model capability is high~\cite{gao2025single}, and a 14-category failure
taxonomy shows modest gains from simple fixes~\cite{cemri2025why}. No published
study cross-varies both architecture and frontier-model family on CUAD under a natural
class distribution with paired statistical tests.

\medskip
\noindent\textbf{Research Questions.}

\noindent\textbf{Main RQ:} Under what conditions do multi-agent architectures improve
contract-analysis accuracy, efficiency, and explainability, and which patterns are most
effective?

\begin{description}[leftmargin=2.5em, labelwidth=2em, itemsep=2pt]
  \item[\textbf{RQ1.}] Which pattern --- pipeline decomposition, specialist routing,
    or consolidated prompting --- achieves the highest clause-extraction $F_2$?
  \item[\textbf{RQ2.}] How do multi-agent configurations compare to single-agent
    baselines in $F_2$ and provider cost across clause categories and contract lengths?
  \item[\textbf{RQ3.}] Which architectural decisions produce auditable reasoning
    traces, and what tradeoffs do they introduce?
\end{description}

\medskip
\noindent\textbf{Hypotheses.}

\begin{description}[leftmargin=2.5em, labelwidth=2em, itemsep=2pt]
  \item[\textbf{H1.}] At least one multi-agent configuration outperforms the
    strongest single-agent baseline on $F_2$ (McNemar exact, $\alpha{=}0.05$,
    BH-corrected).
  \item[\textbf{H2.}] Multi-agent configurations reduce the rare--common $F_2$ gap
    relative to single-agent baselines (paired bootstrap on gap difference, 95\% CI).
  \item[\textbf{H3.}] The LLM-routed specialist system M1 is statistically
    distinguishable from the consolidated specialist ablation M6 on $F_2$
    (two-sided McNemar exact).
\end{description}

\medskip
\noindent\textbf{Contributions.}

\begin{enumerate}[leftmargin=2em, itemsep=2pt]
  \item \textbf{Sampling correction.}  A 30\%/70\% positive/negative sampler for
    CUAD that exposes over-extraction as the dominant zero-shot failure mode.
  \item \textbf{Cross-architecture benchmark.}  $5 \times 11 = 55$ runs across five
    configurations and eleven frontier models on 81 CUAD long contracts,
    yielding 163{,}644 analysed predictions.
  \item \textbf{Failure-mode identification.}  Naming and quantifying the
    \emph{classification-gate rejection cascade} in M0: $70$--$98\%$ of M0 false
    negatives traced to the classifier gate on Gemini-family models.
  \item \textbf{Routing critique.}  Showing M1's LLM router achieves $100\%$
    accuracy on 10/11 models and is statistically indistinguishable from the
    deterministic-lookup ablation M6 ($\Delta F_2 = 0.000$).
  \item \textbf{Practitioner analysis.}  A six-gap analysis mapping CUAD
    performance to enterprise CLM deployment reality.
\end{enumerate}
